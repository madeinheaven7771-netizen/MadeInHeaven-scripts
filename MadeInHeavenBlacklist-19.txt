-- BlacklistServer (Script inside ServerScriptService)

local Players         = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

-- ── RemoteEvents setup ──────────────────────────────────────
local Remotes = Instance.new("Folder")
Remotes.Name  = "MIH_Remotes"
Remotes.Parent = ReplicatedStorage

local RE_Execute  = Instance.new("RemoteEvent")
RE_Execute.Name   = "ExecuteBlacklist"
RE_Execute.Parent = Remotes

local RE_Sync     = Instance.new("RemoteEvent")
RE_Sync.Name      = "SyncBlacklist"
RE_Sync.Parent    = Remotes

-- ── Config ──────────────────────────────────────────────────
local TOOL_NAME = "The Arkenstone"
local ADMINS = {
    -- Put your UserId(s) here so only you can use this:
    -- [1234567] = true,
}
-- Set to true to allow anyone, false to restrict to ADMINS table
local ALLOW_ALL = true

-- ── State ────────────────────────────────────────────────────
local blacklist   = {}   -- { [username] = true }
local modeState   = { mode = "Default", customCmds = {} }

-- ── Helper: is caller authorised ────────────────────────────
local function isAdmin(player)
    if ALLOW_ALL then return true end
    return ADMINS[player.UserId] == true
end

-- ── Helper: give tool to a player ───────────────────────────
local function giveArkenstone(player)
    local char = player.Character
    local bp   = player:FindFirstChild("Backpack")
    if not bp and not char then return end

    -- Remove duplicates first
    for _, v in ipairs(bp and bp:GetChildren() or {}) do
        if v.Name == TOOL_NAME then v:Destroy() end
    end
    if char then
        for _, v in ipairs(char:GetChildren()) do
            if v.Name == TOOL_NAME then v:Destroy() end
        end
    end

    local tool = Instance.new("Tool")
    tool.Name           = TOOL_NAME
    tool.RequiresHandle = false
    tool.CanBeDropped   = false

    -- Visual handle (glowing gem look)
    local handle = Instance.new("Part")
    handle.Name      = "Handle"
    handle.Size      = Vector3.new(0.8, 0.8, 0.8)
    handle.BrickColor = BrickColor.new("Bright yellow")
    handle.Material  = Enum.Material.Neon
    handle.Parent    = tool

    tool.Parent = bp or char
end

-- ── Helper: run commands on a player ────────────────────────
local function runCommands(target, commands)
    for _, cmd in ipairs(commands) do
        local msg = cmd .. " " .. target.Name
        -- Broadcast command text to all clients (shows in their output / can hook to admin)
        RE_Sync:FireAllClients("CMD", msg)
        -- Built-in effects:
        local char = target.Character
        if char then
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChildOfClass("Humanoid")

            local cmdLower = cmd:lower()

            if cmdLower == "freeze" then
                if hrp then
                    hrp.Anchored = true
                end

            elseif cmdLower == "unfreeze" then
                if hrp then
                    hrp.Anchored = false
                end

            elseif cmdLower == "mute" then
                -- Mute via chat tag / property
                local success = pcall(function()
                    game:GetService("Chat"):SetSpeakerExtraData(target.Name, "Muted", true)
                end)
                -- Fallback: set a value others can read
                local muteVal = target:FindFirstChild("MIH_Muted") or Instance.new("BoolValue")
                muteVal.Name   = "MIH_Muted"
                muteVal.Value  = true
                muteVal.Parent = target

            elseif cmdLower == "glitch" then
                -- Rapidly teleport slightly to create glitch effect
                if hrp then
                    task.spawn(function()
                        for i = 1, 20 do
                            hrp.CFrame = hrp.CFrame * CFrame.new(
                                math.random(-3, 3),
                                math.random(0, 2),
                                math.random(-3, 3)
                            )
                            task.wait(0.05)
                        end
                    end)
                end

            elseif cmdLower == "blind" then
                -- Fire to client to apply blur
                RE_Sync:FireClient(target, "BLIND", true)

            elseif cmdLower == "unblind" then
                RE_Sync:FireClient(target, "BLIND", false)

            elseif cmdLower == "kill" then
                if hum then hum.Health = 0 end

            elseif cmdLower == "kick" then
                target:Kick("You have been removed from this server.")

            else
                -- Custom / unknown command: just log and fire to all admins
                RE_Sync:FireAllClients("CMD", "[Custom] " .. msg)
            end
        end
    end
end

-- ── Main remote handler ──────────────────────────────────────
RE_Execute.OnServerEvent:Connect(function(sender, action, data)
    if not isAdmin(sender) then return end

    if action == "ADD" then
        -- data = { username = string }
        local username = tostring(data.username)
        blacklist[username] = true
        -- If they're already in the server, act now
        local target = Players:FindFirstChild(username)
        if target then
            giveArkenstone(target)
            local cmds = modeState.mode == "Default"
                and {"Freeze", "Mute", "Glitch", "Blind"}
                or modeState.customCmds
            runCommands(target, cmds)
        end
        RE_Sync:FireAllClients("LIST", blacklist)

    elseif action == "REMOVE" then
        local username = tostring(data.username)
        blacklist[username] = nil
        RE_Sync:FireAllClients("LIST", blacklist)

    elseif action == "SET_MODE" then
        modeState.mode       = data.mode
        modeState.customCmds = data.customCmds or {}

    elseif action == "EXECUTE_NOW" then
        -- data = { username = string }
        local target = Players:FindFirstChild(tostring(data.username))
        if target then
            giveArkenstone(target)
            local cmds = modeState.mode == "Default"
                and {"Freeze", "Mute", "Glitch", "Blind"}
                or modeState.customCmds
            runCommands(target, cmds)
        end
    end
end)

-- ── Auto-act when blacklisted player joins ───────────────────
Players.PlayerAdded:Connect(function(player)
    if blacklist[player.Name] then
        task.wait(2) -- wait for character to load
        if not player or not player.Parent then return end
        giveArkenstone(player)
        local cmds = modeState.mode == "Default"
            and {"Freeze", "Mute", "Glitch", "Blind"}
            or modeState.customCmds
        runCommands(player, cmds)
    end
end)

print("[MIH Blacklist] Server loaded.")

-- ── Utility ──────────────────────────────────────────────────
local function Corner(p, r)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, r or 8)
    c.Parent = p
    return c
end

local function Stroke(p, col, t)
    local s = Instance.new("UIStroke")
    s.Color = col or C.Border
    s.Thickness = t or 1.5
    s.Parent = s
    s.Parent = p
    return s
end

local function Tween(obj, props, t, style)
    TweenService:Create(obj,
        TweenInfo.new(t or 0.18, style or Enum.EasingStyle.Quad),
        props
    ):Play()
end

local function Hover(btn, n, h)
    btn.MouseEnter:Connect(function()  Tween(btn, {BackgroundColor3 = h}, 0.1) end)
    btn.MouseLeave:Connect(function()  Tween(btn, {BackgroundColor3 = n}, 0.1) end)
end

local function MakeDraggable(frame, handle)
    local drag, ds, sp = false, nil, nil
    handle.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            drag = true
            ds   = i.Position
            sp   = frame.Position
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if drag and i.UserInputType == Enum.UserInputType.MouseMovement then
            local d = i.Position - ds
            frame.Position = UDim2.new(
                sp.X.Scale, sp.X.Offset + d.X,
                sp.Y.Scale, sp.Y.Offset + d.Y
            )
        end
    end)
    UserInputService.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 then
            drag = false
        end
    end)
end

local function SetStatus(msg, col)
    StatusLbl.Text       = msg
    StatusLbl.TextColor3 = col or C.Muted

local function SyncModeToServer()
    RE_Execute:FireServer("SET_MODE", {
        mode       = State.Mode,
        customCmds = State.CustomCmds,
    })
end

local function RefreshTags()
    for _, c in ipairs(TagScroll:GetChildren()) do
        if c:IsA("Frame") then c:Destroy() end
    end
    for _, cmd in ipairs(State.CustomCmds) do
        local Tag = Instance.new("Frame")
        Tag.Size             = UDim2.new(0, 0, 0, 22)
        Tag.AutomaticSize    = Enum.AutomaticSize.X
        Tag.BackgroundColor3 = C.TagBG
        Tag.BorderSizePixel  = 0
        Tag.ZIndex           = 14
        Tag.Parent           = TagScroll
        Corner(Tag, 5)
        Stroke(Tag, C.Accent, 1)

        local Pad = Instance.new("UIPadding")
        Pad.PaddingLeft  = UDim.new(0, 6)
        Pad.PaddingRight = UDim.new(0, 20)
        Pad.Parent = Tag

        local TLbl = NewLabel(Tag, cmd, 10, Enum.Font.GothamSemibold, Enum.TextXAlignment.Left, C.Text, 15)
        TLbl.Size         = UDim2.new(0, 0, 1, 0)
        TLbl.AutomaticSize = Enum.AutomaticSize.X

        local Del = NewButton(Tag, "✕", 8, Enum.Font.GothamBold, Color3.new(0,0,0), C.Muted, 16)
        Del.Size             = UDim2.new(0, 14, 0, 14)
        Del.Position         = UDim2.new(1, -17, 0.5, -7)
        Del.BackgroundTransparency = 1

        local cap = cmd
        Del.MouseButton1Click:Connect(function()
            for i, v in ipairs(State.CustomCmds) do
                if v == cap then table.remove(State.CustomCmds, i) break end
            end
            RefreshTags()
            SyncModeToServer()
        end)
    end
    TagScroll.CanvasSize = UDim2.new(0, 0, 0, math.ceil(#State.CustomCmds / 4) * 26)
end

local function RefreshList()
    for _, c in ipairs(ListScroll:GetChildren()) do
        if c:IsA("Frame") then c:Destroy() end
    end

    local n = 0
    for _ in pairs(State.Blacklisted) do n = n + 1 end
    CountLbl.Text = n .. (n == 1 and " player" or " players")

    if n == 0 then
        EmptyLbl.Visible     = true
        ListScroll.CanvasSize = UDim2.new(0, 0, 0, 0)
        return
    end
    EmptyLbl.Visible = false

    local order = 0
    for username in pairs(State.Blacklisted) do
        order = order + 1

        local Rw = Instance.new("Frame")
        Rw.Size             = UDim2.new(1, -4, 0, 30)
        Rw.BackgroundColor3 = (order % 2 == 0) and C.RowEven or C.BG
        Rw.BorderSizePixel  = 0
        Rw.LayoutOrder      = order
        Rw.ZIndex           = 14
        Rw.Parent           = ListScroll
        Corner(Rw, 6)

        local Dot = Instance.new("Frame")
        Dot.Size             = UDim2.new(0, 7, 0, 7)
        Dot.Position         = UDim2.new(0, 9, 0.5, -3.5)
        Dot.BackgroundColor3 = C.Red
        Dot.BorderSizePixel  = 0
        Dot.ZIndex           = 15
        Dot.Parent           = Rw
        Corner(Dot, 3)

        local NLbl = NewLabel(Rw, username, 11, Enum.Font.GothamSemibold, Enum.TextXAlignment.Left, C.Text, 15)
        NLbl.Size     = UDim2.new(0.55, 0, 1, 0)
        NLbl.Position = UDim2.new(0, 22, 0, 0)

        -- Execute now button
        local ExBtn = NewButton(Rw, "▶", 10, Enum.Font.GothamBold, C.Accent, C.Text, 15)
        ExBtn.Size     = UDim2.new(0, 24, 0, 20)
        ExBtn.Position = UDim2.new(1, -82, 0.5, -10)
        Corner(ExBtn, 5)
        Hover(ExBtn, C.Accent, C.AccentD)

        local RmBtn = NewButton(Rw, "Remove", 9, Enum.Font.GothamBold, C.Red, C.White, 15)
        RmBtn.Size     = UDim2.new(0, 52, 0, 20)
        RmBtn.Position = UDim2.new(1, -56, 0.5, -10)
        Corner(RmBtn, 5)
        Hover(RmBtn, C.Red, C.RedD)

        local cap = username
        ExBtn.MouseButton1Click:Connect(function()
            RE_Execute:FireServer("EXECUTE_NOW", { username = cap })
            SetStatus("Executed on: " .. cap, C.Red)
        end)
        RmBtn.MouseButton1Click:Connect(function()
            RE_Execute:FireServer("REMOVE", { username = cap })
            State.Blacklisted[cap] = nil
            RefreshList()
            SetStatus("Removed: " .. cap, C.Green)
        end)
    end

    ListScroll.CanvasSize = UDim2.new(0, 0, 0, n * 33 + 4)
end

local function SetMode(mode)
    State.Mode = mode
    if mode == "Default" then
        BtnDef.BackgroundColor3  = C.Accent
        BtnDef.TextColor3        = C.Text
        BtnCust.BackgroundColor3 = C.BGAlt
        BtnCust.TextColor3       = C.Muted
        R3.Visible = false
        R3.Size    = UDim2.new(1, 0, 0, 0)
        SetStatus("Mode: Default", C.Muted)
    else
        BtnCust.BackgroundColor3 = C.Accent
        BtnCust.TextColor3       = C.Text
        BtnDef.BackgroundColor3  = C.BGAlt
        BtnDef.TextColor3        = C.Muted
        R3.Visible = true
        R3.Size    = UDim2.new(1, 0, 0, 118)
        RefreshTags()
        SetStatus("Mode: Custom", C.Muted)
    end
    SyncModeToServer()
end

BtnDef.MouseButton1Click:Connect(function()  SetMode("Default") end)
BtnCust.MouseButton1Click:Connect(function() SetMode("Custom")  end)

-- ── Add custom command ───────────────────────────────────────
AddCmdBtn.MouseButton1Click:Connect(function()
    local cmd = CmdInput.Text:match("^%s*(.-)%s*$")
    if cmd == "" then return end
    if #State.CustomCmds >= 10 then
        SetStatus("Max 10 commands!", C.Red) return
    end
    for _, v in ipairs(State.CustomCmds) do
        if v:lower() == cmd:lower() then
            SetStatus("Already added.", C.Red) return
        end
    end
    table.insert(State.CustomCmds, cmd)
    CmdInput.Text = ""
    RefreshTags()
    SyncModeToServer()
    SetStatus("Added cmd: " .. cmd, C.Green)
end)

-- ── Toggle blacklist ─────────────────────────────────────────
TogBtn.MouseButton1Click:Connect(function()
    State.Enabled = not State.Enabled
    if State.Enabled then
        TogBtn.BackgroundColor3 = C.Green
        TogBtn.Text             = "ON"
        SetStatus("Blacklist ON", C.Green)
    else
        TogBtn.BackgroundColor3 = C.Red
        TogBtn.Text             = "OFF"
        SetStatus("Blacklist OFF", C.Muted)

-- ── Add player ───────────────────────────────────────────────
AddPlrBtn.MouseButton1Click:Connect(function()
    local name = PlrInput.Text:match("^%s*(.-)%s*$")
    if name == "" then SetStatus("Enter a username.", C.Red) return end
    if State.Blacklisted[name] then SetStatus(name .. " already listed.", C.Red) return end
    State.Blacklisted[name] = true
    PlrInput.Text = ""
    RefreshList()
    RE_Execute:FireServer("ADD", { username = name })
    SetStatus("Blacklisted: " .. name, C.Red)
end)

-- ── Sync back from server ─────────────────────────────────────
RE_Sync.OnClientEvent:Connect(function(eventType, data)
    if eventType == "LIST" then
        State.Blacklisted = data
        RefreshList()
    elseif eventType == "CMD" then
        SetStatus(tostring(data), C.Red)
    elseif eventType == "BLIND" then
        -- Apply blur effect to this client
        local blur = game:GetService("Lighting"):FindFirstChild("MIH_Blur")
        if data == true then
            if not blur then
                blur = Instance.new("BlurEffect")
                blur.Name  = "MIH_Blur"
                blur.Size  = 56
                blur.Parent = game:GetService("Lighting")
            end
        else
            if blur then blur:Destroy() end
        end
    end
end)

  In Roblox Studio, open ServerScriptService.
  Create a new Script named "BlacklistServer".
  Paste FILE 1 code into it.

STEP 2:
  Open StarterPlayer > StarterPlayerScripts.
  Create a new LocalScript named "BlacklistUI".
  Paste FILE 2 code into it.

STEP 3 (Optional - Admin Lock):
  In the server script (FILE 1), find the ADMINS table:
      local ADMINS = {}
  Add your UserId like this:
      local ADMINS = { [123456789] = true }
  Then set:
      local ALLOW_ALL = false
  This restricts the UI to only work for that UserId.

BUILT-IN COMMANDS (Default Mode):
  Freeze   — Anchors the player's HumanoidRootPart (can't move)
  Mute     — Sets a MIH_Muted BoolValue on the player
  Glitch   — Rapidly teleports player around for 1 second
  Blind    — Applies a BlurEffect via Lighting to target's client

CUSTOM COMMANDS:
  Any word you type becomes a command sent as:
      "[word] [username]"
  Built-in names (freeze/mute/glitch/blind/kill/kick/unfreeze/unblind)
  are handled automatically. All others are printed and fired to
  all admin clients via the status bar.

  The ▶ button next to each blacklisted player fires all
  commands on them immediately, even if they're already in server.
